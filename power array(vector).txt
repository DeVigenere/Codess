#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>

using namespace std;

int main() {
	int n, count = 0;
	double res = 0;
	cout << "Enter how many numbers (1 to 100): " << endl;
	cin >> n;
	if (n < 1 || n>100) {
		cout << "Enter correct numbers next time!" << endl;
		return 0;
	}
	vector <int> vec(n);
	for (int i = 0; i < n; i++) {
		cout << "Enter a value (0 to 1000): " << endl;
		cin >> vec[i];
		if (vec[i] < 0 || vec[i] > 1000) {
			cout << "Enter correct numbers next time!" << endl;
			return 0;
		}
	}
	sort(vec.begin(), vec.end());
	for (int i = 0; i < n; i++) {
		if (vec[i] == vec[n - 1]) {
			count++;
		}
	}
	res = floor(((double)count / (double)n) * 100.);
	printf("%.0f%%", res);
}
