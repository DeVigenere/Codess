#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n;
	
	int count1 = 0, count2 = 0, count3 = 0;
	
	cout << "Size of vector: " << endl;
	cin >> n;
	
	vector<int> vec(n);
	
	for (int i = 0; i < n; i++) {
		cout << "Enter from 1 to 3: " << endl;
		cin >> vec[i];
		if (vec[i] != 1 && vec[i] != 2 && vec[i] != 3) {
			cout << "Error." << endl;
			return 0;
		}
	}
	
	for (int i = 0; i < n; i++) {
		if (vec[i] == 1) {
			count1++;
		}
		else if (vec[i] == 2) {
			count2++;
		}
		else if (vec[i] == 3) {
			count3++;
		}
	}
	if (count1 < count2 && count2 < count3) {
		cout << count1 + count2;
	}
	else if (count1 < count3 && count2 < count1) {
		cout << count1 + count2;
	}
	else if (count2 < count1 && count3 < count1) {
		cout << count2 + count3;
	}
	else if (count2 < count3 && count3 < count2) {
		cout << count2 + count3;
	}
	else if (count1 < count2 && count3 < count2) {
		cout << count1 + count3;
	}
	
}
