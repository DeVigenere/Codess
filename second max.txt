#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int n;
	cin >> n;
	
	vector<int> vec(n);
	
	for (int i = 0; i < n; i++) {
		cin >> vec[i];
	}

	if (n != 0) {
		sort(vec.begin(), vec.end());
	}
	
	if (n > 1) {
		cout << vec[n - 2] << endl;
	}
	else {
		cout << "-1" << endl;
	}
}
