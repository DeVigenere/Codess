#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
	int n;
	cin >> n;
	if (n < 0 || n > 100) {
		cout << "Wrong enter!";
		return 0;
	}
	vector <int> vec(n);
	
	for (int i = 0; i < n; i++) {
		cin >> vec[i];
	}
	if (n > 1) {
		sort(vec.begin(), vec.end());

		for (int i = 0; i < n; i++) {
			if (vec[i] == vec[n - 1]) {
				vec[i] = 0;
			}
		}

		sort(vec.begin(), vec.end());
		
	}
	if (vec[n-1] == 0 || n < 2){
		cout << "-1";
	}
	else {
		cout << vec[n - 1];
	}
	return 0;
}
