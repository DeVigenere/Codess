#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
	int n;
	int count = 0;
	
	cout << "Enter now many numbers: " << endl;
	cin >> n;
	
	vector<int> vec(n);
	
	for (int i = 0; i < n; i++) {
		cin >> vec[i];
	}
	sort(vec.begin(), vec.end());
	
	for (int i = 0; i < n; i++) {
		if (vec[n - 1] == vec[i]) {
			count++;
		}
	}
	
	cout << vec[n - 1] << " " << count << endl;

}
