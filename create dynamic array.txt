#include <iostream>

using namespace std;

const int rows = 3;
const int cols = 4;

int main() {
	int** arr = new int* [rows];
	for (int i = 0; i < rows; i++) {
		arr[i] = new int[cols];
	}
	for (int i = 0; i < rows; i++) {
		cout << endl;
		for (int j = 0; j < cols; j++) {
			arr[i][j] = i * i;
			cout << arr[i][j] << " ";
		}
	}
	for (int i = 0; i < rows; i++) {
		delete[] arr[i];
		arr[i] = nullptr;
	}
	delete[] arr;
}
